

CREATE TABLE hr_user (
  pkid number NOT NULL,
  stat nvarchar2(40) DEFAULT NULL,
  created_date date DEFAULT NULL,
  created_user number DEFAULT NULL,
  UserLoginName nvarchar2(150) NOT NULL,
  userLoginEncry nvarchar2(150) DEFAULT NULL,
  UserFullName nvarchar2(45) DEFAULT NULL,
  password nvarchar2(150) DEFAULT NULL,
  Notes nvarchar2(100) DEFAULT NULL,
  branch_id number DEFAULT NULL,
  PRIMARY KEY (pkid)
 
  
) ;

CREATE TABLE hr_list_type (
  PKID number NOT NULL,
  LIST_NAME nvarchar2(100) DEFAULT NULL,
  SELECT_STATEMENT nvarchar2(100) DEFAULT NULL,
  PRIMARY KEY (PKID)
) ;

CREATE TABLE hr_forms_hd (
  PKID number NOT NULL,
  AR_NAME nvarchar2(145) DEFAULT NULL,
  EN_NAME nvarchar2(145) DEFAULT NULL,
  MULTI_FORM_FOR_USER char(1) DEFAULT NULL,
  FORM_TYPE_EN nvarchar2(128) DEFAULT NULL,
  FORM_TYPE_AR nvarchar2(128) DEFAULT NULL,
  PRIMARY KEY (PKID)
) ;

CREATE TABLE hr_forms_de (
  PKID number NOT NULL,
  HEADER_ID number DEFAULT NULL,
  CONTROL_TYPE nvarchar2(100) DEFAULT NULL,
  EN_NAME nvarchar2(200) DEFAULT NULL,
  AR_NAME nvarchar2(200) DEFAULT NULL,
  LIST_ID number DEFAULT NULL,
  FREE_LIST_NAME nvarchar2(100) DEFAULT NULL,
  PROPERITY_NOTE nvarchar2(300) DEFAULT NULL,
  PRIMARY KEY (PKID),
  CONSTRAINT Form_de_Form_hd_fk FOREIGN KEY (HEADER_ID) REFERENCES hr_forms_hd (PKID)
) ;


CREATE TABLE hr_templet_header (
  PKID number NOT NULL,
  CREATED_USER number DEFAULT NULL,
  CREATED_DATE date DEFAULT NULL,
  TEMPLET_NAME nvarchar2(100) DEFAULT NULL,
  FORM_ENG_NAME nvarchar2(100) DEFAULT NULL,
  FORM_ARABIC_NAME nvarchar2(100) DEFAULT NULL,
  MULTI_FORM_FOR_USER char(1) DEFAULT NULL,
  FORM_TYPE_EN nvarchar2(45) DEFAULT NULL,
  FORM_TYPE_AR nvarchar2(45) DEFAULT NULL,
  PRIMARY KEY (PKID)
) ;


CREATE TABLE hr_templet_details (
  PKID number NOT NULL,
  HEADER_ID number DEFAULT NULL,
  CONTROL_TYPE nvarchar2(100) DEFAULT NULL,
  EN_NAME nvarchar2(100) DEFAULT NULL,
  AR_NAME nvarchar2(100) DEFAULT NULL,
  CONTROL_VALUE nvarchar2(1000) DEFAULT NULL,
  CREATED_USER number DEFAULT NULL,
  CREATED_DATE date DEFAULT NULL,
  REAL_VALUE nvarchar2(45) DEFAULT NULL,
  PRIMARY KEY (PKID),
 
  CONSTRAINT template_details_fk FOREIGN KEY (HEADER_ID) REFERENCES HR_templet_header (PKID)
) ;


CREATE TABLE hr_user_templet (
  PKID number NOT NULL,
  USER_ID number DEFAULT NULL,
  TEMPLET_ID number DEFAULT NULL,
  PRIMARY KEY (PKID),

  CONSTRAINT UserTemp_templateHD_fk FOREIGN KEY (TEMPLET_ID) REFERENCES hr_templet_header (PKID),
  CONSTRAINT UserTemp_user_Pk FOREIGN KEY (USER_ID) REFERENCES HR_user (pkid)
) ;


CREATE TABLE hr_updatehistory (
  pk number NOT NULL,
  created_user number DEFAULT NULL,
  Created_date date DEFAULT NULL,
  TABLE_NAME nvarchar2(100) DEFAULT NULL,
  COLUMN_NAME nvarchar2(100) DEFAULT NULL,
  ROW_PKID number DEFAULT NULL,
  OLD_VALUE nvarchar2(1000) DEFAULT NULL,
  New_value nvarchar2(1000) DEFAULT NULL,
  ARABIC_NAME nvarchar2(200) DEFAULT NULL,
  PRIMARY KEY (pk)
) ;
